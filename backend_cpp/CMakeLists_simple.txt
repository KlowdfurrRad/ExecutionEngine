cmake_minimum_required(VERSION 3.10)
project(CashFuturesTHV)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find required packages
if(WIN32)
    # Windows specific settings
    add_definitions(-D_WIN32_WINNT=0x0601)
endif()

# Add executable
add_executable(cash_futures_thv_simple main_simple.cpp)

# Link libraries
if(WIN32)
    target_link_libraries(cash_futures_thv_simple ws2_32)
else()
    # Linux/Mac - no additional libraries needed for standard sockets
endif()

# Compiler-specific options
if(MSVC)
    target_compile_options(cash_futures_thv_simple PRIVATE /W4)
else()
    target_compile_options(cash_futures_thv_simple PRIVATE -Wall -Wextra -pedantic)
endif()

# Set output directory
set_target_properties(cash_futures_thv_simple PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
